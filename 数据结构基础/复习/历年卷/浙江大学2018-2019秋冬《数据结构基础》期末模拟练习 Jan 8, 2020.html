bool IsTopSeq( Vertex Seq[], LGraph G ) {
	int Degree[1000];
	for (int i = 0; i <= G->N_v; i++) {
		Degree[i] = 0;
	}
	PtrToAdjVNode tnode;
	for (int i = 0; i < G->N_v; i++) {
		tnode = G->G[i].FirstEdge;
		while (tnode) {
			Degree[tnode->AdjV]++;
			tnode = tnode->Next;
		}
	}
	for (int i = 0; i < G->N_v; i++) {
		if (Degree[Seq[i] - 1] == 0) {
			tnode = G->G[Seq[i] - 1].FirstEdge;
			while (tnode) {
				Degree[tnode->AdjV]--;
				tnode = tnode->Next;
			}
		}
		else
			return false;
	}
	return true;
}
